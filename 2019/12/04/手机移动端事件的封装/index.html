# 手机移动端touch事件封装

# 一、背景

1、因为在移动端click事件会有300ms 的延迟,用户体验不好
2、touch事件封装，便于使用

# 二、代码

(function(window) {
  function myMobile(ele) {
    return myMobile.prototype._init(ele);
  }
  myMobile.prototype = {
    _init: function(ele) {
      if (typeof ele === "string") {
        this.ele = window.document.querySelector(ele);
      }
      return this;
    },
    // 触摸事件
    tap: function(handler) {
      var starTime, endTime;
      function touchFn(e) {
        e.preventDefault();
        switch (e.type) {
          case "touchstart":
            starTime = Date.now();
            break;
          case "touchend":
            endTime = Date.now();
            if (endTime - starTime < 300) {
              handler.call(this, e);
            }
            break;
        }
      }
      this.ele.addEventListener("touchstart", touchFn);
      this.ele.addEventListener("touchend", touchFn);
    },
    //长按事件
    longPress: function(handler) {
      this.ele.addEventListener("touchstart", touchFn);
      this.ele.addEventListener("touchmove", touchFn);
      this.ele.addEventListener("touchend", touchFn);
      var timerId;
      function touchFn(e) {
        switch (e.type) {
          case "touchstart":
            timerId = setTimeout(function() {
              handler.call(this, e);
            }, 500);
            break;
          case "touchmove":
            clearInterval(timerId);
            break;
          case "touchend":
            clearTimeout(timerId);
            break;
        }
      }
    },
    // 左滑事件
    slideLeft: function(handler) {
      this.ele.addEventListener("touchstart", touchFn);
      this.ele.addEventListener("touchend", touchFn);
      var startX, startY, endX, endY;
      function touchFn(e) {
        var firstTouch = e.changedTouches[0];
        switch (e.type) {
          case "touchstart":
            startX = firstTouch.pageX;
            startY = firstTouch.pageY;
            break;
          case "touchend":
            endX = firstTouch.pageX;
            endY = firstTouch.pageY;
            if (
              Math.abs(endX - startX) >= Math.abs(endY - startY) &&
              startX - endX > 25
            ) {
              handler.call(this, e);
            }
            break;
        }
      }
    },
    // 右滑事件
    slideRight: function(handler) {
      this.ele.addEventListener("touchstart", touchFn);
      this.ele.addEventListener("touchend", touchFn);
      var startX, startY, endX, endY;
      function touchFn(e) {
        var firstTouch = e.changedTouches[0];
        switch (e.type) {
          case "touchstart":
            startX = firstTouch.pageX;
            startY = firstTouch.pageY;
            break;
          case "touchend":
            endX = firstTouch.pageX;
            endY = firstTouch.pageY;
            if (
              Math.abs(endX - startX) >= Math.abs(endY - startY) &&
              endX - startX > 25
            ) {
              handler.call(this, e);
            }
            break;
        }
      }
    },
    //跟随事件
    follow: function() {
      this.ele.addEventListener("touchstart", touchFn);
      this.ele.addEventListener("touchmove", touchFn);
      var currentLeft, currentTop, startX, startY, moveX, moveY;
      function touchFn(event) {
        switch (event.type) {
          case "touchstart":
            currentLeft = parseInt(this.offsetLeft);
            currentTop = parseInt(this.offsetTop);
            startX = event.touches[0].pageX;
            startY = event.touches[0].pageY;
            break;
          case "touchmove":
            moveX = event.touches[0].pageX;
            moveY = event.touches[0].pageY;
            this.style.marginLeft = currentLeft + (moveX - startX) + "px";
            this.style.marginTop = currentTop + (moveY - startY) + "px";
            break;
        }
      }
    }
  };
  window.$ = window.myMobile = myMobile;
})(window);
